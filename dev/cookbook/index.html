import Yota: YotaRuleConfig, rrule_via_ad

h(w, b, x) = w * x .+ b

w, b, x = rand(3, 4), rand(3), rand(4, 5)
val, pb = rrule_via_ad(YotaRuleConfig(), h, w, b, x)
